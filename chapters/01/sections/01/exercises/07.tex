\begin{exercise}
Check that the following are (trivial) examples of fibrations
\begin{equation*}
\begin{tikzcd}[%
  , row sep=scriptsize
  , column sep=large
  ]
\cat{B} \times \cat{C} \arrow[d, "\fst"]
& \cat{B} \arrow[d, "\id"]
& \cat{B} \arrow[d]
&[-4.2em]
&[-2em] X \arrow[d]
\\
\cat{B}
& \cat{B}
& 1
& = \{*\}
& I
\end{tikzcd}
\end{equation*}
where \(X\), \(I\) are sets (\emph{i.e.} discrete categories).
\end{exercise}
